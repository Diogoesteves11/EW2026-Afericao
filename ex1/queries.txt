// 1. Quantos automóveis foram reparados?
db.repairs.distinct("viatura.matricula").length


// 2. Quantos automóveis da marca "Cadillac" foram reparados?
db.repairs.distinct("viatura.matricula", { "viatura.marca": "Cadillac" }).length


// 3. Que marcas de automóveis já apareceram na oficina (ordenada alfabeticamente e sem repetições)?
db.repairs.distinct("viatura.marca").sort()


// 4. Qual a distribuição por tipo de reparação? 
db.repairs.aggregate([
  { $unwind: "$intervencoes" },
  { $group: {
      _id: "$intervencoes.nome",
      total: { $sum: 1 }
    }
  },
  { $sort: { total: -1 } } 
])


// 5. Qual a distribuição do número de clientes por ano?
db.repairs.aggregate([
  { 
    $group: {
      _id: { $substr: ["$data", 0, 4] },
      clientes_unicos: { $addToSet: "$nif" }
    }
  },
  { 
    $project: {
      _id: 0,
      ano: "$_id",
      total_clientes: { $size: "$clientes_unicos" } 
    }
  },
  { $sort: { ano: 1 } } 
])
